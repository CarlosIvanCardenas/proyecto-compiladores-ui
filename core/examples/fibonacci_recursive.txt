fun fibonacci(n: int): int {
    var val: int
    if (n <= 0) {
        write("Incorrect input")
        val = 0
    } else {
        if ( n == 1) {
            val = 0
        } else {
            if (n == 2) {
                val = 1
            } else {
                val = fibonacci(n-1) + fibonacci(n-2)
            }
        }
    }
    return (val)
}
